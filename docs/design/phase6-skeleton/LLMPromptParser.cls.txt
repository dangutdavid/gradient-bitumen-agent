public with sharing class LLMPromptParser implements AgentPromptParser {

    public AgentParseResult parse(String text, AgentContext ctx) {
        // PER-TURN fallback lives INSIDE this class: it handles "the model call failed
        // right now" (callout error, timeout, unusable/empty response) by degrading to
        // the proven rule-based parser for THIS turn. This is distinct from
        // AgentParserFactory's class-missing fallback, which handles "the configured
        // parser class doesn't exist" (misconfiguration). Both safety nets matter.
        try {
            Agent_Setting__mdt cfg = Agent_Setting__mdt.getInstance('Default');
            String provider = cfg != null ? cfg.LLM_Provider__c : 'SalesforceModels'; // Anthropic|OpenAI|SalesforceModels
            Integer timeoutMs = (cfg != null && cfg.LLM_Timeout_Ms__c != null)
                ? Integer.valueOf(cfg.LLM_Timeout_Ms__c) : 10000;

            String prompt = buildPrompt(text, ctx);
            String rawJson = callModel(provider, prompt, timeoutMs);

            if (String.isBlank(rawJson)) {
                return fallback(text, ctx); // unusable/empty -> rule-based this turn
            }
            AgentParseResult res = mapToParseResult(rawJson, ctx);
            return res != null ? res : fallback(text, ctx);
        } catch (Exception e) {
            // CalloutException, timeout, JSON/shape error, anything -> never break a turn.
            return fallback(text, ctx);
        }
    }

    private AgentParseResult fallback(String text, AgentContext ctx) {
        return new RuleBasedPromptParser().parse(text, ctx);
    }
